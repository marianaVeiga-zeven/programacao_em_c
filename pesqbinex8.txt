#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#include <conio.h>
#include <math.h>
#include <string.h>
int main()
{
    setlocale(LC_ALL,"Portuguese");

    struct reg_pessoa{
      char email[40];
      char nome[30];
      char tel[30];
    };
    struct reg_pessoa pessoa[10], aux;
    int i=0, j, ini, fim, meio, achou;
    char pesq[30];

    for(i=0; i<10; i++){
        printf("\nInforme o nome: ");
        gets(pessoa[i].nome);
        setbuf(stdin,NULL);
        printf("Informe o telefone: ");
        gets(pessoa[i].tel);
        setbuf(stdin,NULL);
        printf("Informe o email: ");
        gets(pessoa[i].email);
        setbuf(stdin,NULL);
    }
     for (i=0; i<9; i++){
      for (j=i+1; j<10; j++){
        if (strcmp(pessoa[i].nome,pessoa[j].nome) == 1){
           aux = pessoa[i];
           pessoa[i] = pessoa[j];
           pessoa[j] = aux;
        }
      }
    }
     while(1){
      printf("\nInforme nome para pesquisar - Fim - Encerra:");
      gets(pesq);
      setbuf(stdin,NULL);
      if (strcmp(pesq,"fim") == 0)
         break;
      ini = 0;
      fim = 4;
      achou = 0;
      while (ini <= fim && !achou){
        meio = (ini + fim) / 2;
         if (strcmp(pesq,pessoa[meio].nome) == 0)
            achou = 1;
         else
            if (strcmp(pesq,pessoa[meio].nome) == 1)
               ini = meio + 1;
            else
               fim = meio - 1;
      }
     if(achou)
        printf("\nNome: %s \nTelefone: %s \nEmail: %s ",pesq, pessoa[meio].tel,pessoa[meio].email);
     else
        printf("\n%s nÃ£o encontrado(a)!", pesq);
    }
    return 0;
}
